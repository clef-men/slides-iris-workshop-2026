          \Forall \v{ctx}\ \v{scope}. \\

          \iWpVertical{
            \v{task}\ \v{ctx}

            \zooVal

            \l{pool.context}\ \v{pool}\ \v{ctx}\ \v{scope} \iSep \\

          }
        \end{inboxIndent}
      }{
        \c{async}\ \v{ctx}\ \v{task}
      }[
        t
      ]{
        \l{pool.context}\ \v{pool}\ \v{ctx}\ \v{scope} \iSep \\

        \consumer\ t\ \iPredTwo
      }
    \and
      \iSpec[
        lab=wait-spec
      ]{
        \l{pool.context}\ \v{pool}\ \v{ctx}\ \v{scope} \iSep \\

      }{
        \c{wait}\ \v{ctx}\ t
      }[
        \zooVal
      ]{
        \iCredit{2} \iSep \\

      }
